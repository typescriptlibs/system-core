Class OBJECT

The base Object class for all other classes. Internally defined in upper case to avoid conflicts with reserved keywords.

import { Object, OBJECT } from '@tsl/system-core';
let obj = new Object();
console.log(obj.getType().fullName); // @tsl/system-core/System.Object
obj = new OBJECT();
console.log(obj.getType().fullName); // @tsl/system-core/System.Object
